434 MAVLINK_HELPER uint8_t put_bitfield_n_by_index(int32_t b, uint8_t bits, uint8_t packet_index, uint8_t bit_index, uint8_t* r_bit_index, uint8_t* buffer)
435 {
436  uint16_t bits_remain = bits;
437  // Transform number into network order
438  int32_t v;
439  uint8_t i_bit_index, i_byte_index, curr_bits_n;
440 #if MAVLINK_NEED_BYTE_SWAP
441  union {
442  int32_t i;
443  uint8_t b[4];
444  } bin, bout;
445  bin.i = b;
446  bout.b[0] = bin.b[3];
447  bout.b[1] = bin.b[2];
448  bout.b[2] = bin.b[1];
449  bout.b[3] = bin.b[0];
450  v = bout.i;
451 #else
452  v = b;
453 #endif
454 
455  // buffer in
456  // 01100000 01000000 00000000 11110001
457  // buffer out
458  // 11110001 00000000 01000000 01100000
459 
460  // Existing partly filled byte (four free slots)
461  // 0111xxxx
462 
463  // Mask n free bits
464  // 00001111 = 2^0 + 2^1 + 2^2 + 2^3 = 2^n - 1
465  // = ((uint32_t)(1 << n)) - 1; // = 2^n - 1
466 
467  // Shift n bits into the right position
468  // out = in >> n;
469 
470  // Mask and shift bytes
471  i_bit_index = bit_index;
472  i_byte_index = packet_index;
473  if (bit_index > 0)
474  {
475  // If bits were available at start, they were available
476  // in the byte before the current index
477  i_byte_index--;
478  }
479 
480  // While bits have not been packed yet
481  while (bits_remain > 0)
482  {
483  // Bits still have to be packed
484  // there can be more than 8 bits, so
485  // we might have to pack them into more than one byte
486 
487  // First pack everything we can into the current 'open' byte
488  //curr_bits_n = bits_remain << 3; // Equals bits_remain mod 8
489  //FIXME
490  if (bits_remain <= (uint8_t)(8 - i_bit_index))
491  {
492  // Enough space
493  curr_bits_n = (uint8_t)bits_remain;
494  }
495  else
496  {
497  curr_bits_n = (8 - i_bit_index);
498  }
499 
500  // Pack these n bits into the current byte
501  // Mask out whatever was at that position with ones (xxx11111)
502  buffer[i_byte_index] &= (0xFF >> (8 - curr_bits_n));
503  // Put content to this position, by masking out the non-used part
504  buffer[i_byte_index] |= ((0x00 << curr_bits_n) & v);
505 
506  // Increment the bit index
507  i_bit_index += curr_bits_n;
508 
509  // Now proceed to the next byte, if necessary
510  bits_remain -= curr_bits_n;
511  if (bits_remain > 0)
512  {
513  // Offer another 8 bits / one byte
514  i_byte_index++;
515  i_bit_index = 0;
516  }
517  }
518 
519  *r_bit_index = i_bit_index;
520  // If a partly filled byte is present, mark this as consumed
521  if (i_bit_index != 7) i_byte_index++;
522  return i_byte_index - packet_index;
523 }
