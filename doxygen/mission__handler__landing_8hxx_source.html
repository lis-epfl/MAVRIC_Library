00104 template <class T1, class T2>
00105 Mission_handler::update_status_t Mission_handler_landing<T1, T2>::update()
00106 {
00107     /*****************************
00108     Handle internal landing states 
00109     *****************************/
00110     // Determine if we should switch between the landing states
00111     bool next_state = false;
00112     if (auto_landing_behavior_ == DESCENT_TO_GND)
00113     {
00114         alt_lpf_ = LPF_gain_ * (alt_lpf_) + (1.0f - LPF_gain_) * ins_.position_lf()[Z];
00115         if ((ins_.position_lf()[Z] > -0.1f) && (maths_f_abs(ins_.position_lf()[Z] - alt_lpf_) <= 0.2f))
00116         {
00117             next_state = true;
00118         }
00119     }
00120     else if (auto_landing_behavior_ == DESCENT_TO_SMALL_ALTITUDE)
00121     {
00122         float z_lim = desc_to_ground_altitude_- desc_to_ground_range_;
00123         if ((ins_.position_lf()[Z]) > z_lim)                                         // Drone is below desc2gnd waypoint (for cases when we start landing below it)
00124         {
00125             next_state = true;
00126         }
00127     }
00128     
00129     /*************
00130     Set controller 
00131     *************/
00132     bool ret = false;
00133     switch (auto_landing_behavior_)
00134     {
00135         case DESCENT_TO_SMALL_ALTITUDE:
00136             ret = set_desc_to_small_alt_control_command();
00137             break;
00138 
00139         case DESCENT_TO_GND:
00140             ret = set_desc_to_ground_control_command();
00141             break;
00142     }
00143 
00144     // If we are switching between states, ... then switch
00145     if (next_state)
00146     {
00147         switch (auto_landing_behavior_)
00148         {
00149             case DESCENT_TO_SMALL_ALTITUDE:
00150                 print_util_dbg_print("Automatic-landing: descent_to_GND\r\n");
00151                 auto_landing_behavior_ = DESCENT_TO_GND;
00152                 state_.mav_mode_custom &= static_cast<Mav_mode::custom_mode_t>(0xFFFFFFE0);
00153                 state_.mav_mode_custom |= Mav_mode::CUST_DESCENT_TO_GND;
00154                 alt_lpf_ = ins_.position_lf()[Z];
00155                 break;
00156 
00157             case DESCENT_TO_GND:
00158                 print_util_dbg_print("Auto-landing: disarming motors \r\n");
00159                 auto_landing_behavior_ = DESCENT_TO_SMALL_ALTITUDE;
00160                 //Do not reset custom flag here, to be able to check after landing
00161                 // in case something went wrong. Is reset while arming
00162                 state_.set_armed(false);
00163                 state_.mav_state_ = MAV_STATE_STANDBY;
00164                 is_landed_ = true;
00165                 break;
00166         }
00167     }
00168     
00169 
00170     /********************
00171     Determine status code 
00172     ********************/
00173     if (waypoint_.autocontinue() == 1 && is_landed_)
00174     {
00175         return MISSION_FINISHED;
00176     }
00177 
00178     // Handle control command failed status
00179     if (!ret)
00180     {
00181         return MISSION_FAILED;
00182     }
00183 
00184     return MISSION_IN_PROGRESS;
00185 }
